const express = require('express'); const multer = require('multer'); const fs = require('fs'); const path = require('path'); const app = express(); const PORT = 3000; const upload = multer({ dest: 'public/uploads/' }); app.use(express.static('public')); app.use(express.urlencoded({ extended: true })); const usersPath = './data/users.json'; const artworksPath = './data/artworks.json'; app.post('/create-profile', (req, res) => { const name = req.body.artistName; const users = JSON.parse(fs.readFileSync(usersPath, 'utf-8')); if (!users.includes(name)) users.push(name); fs.writeFileSync(usersPath, JSON.stringify(users, null, 2)); res.redirect('/upload.html'); }); app.post('/upload', upload.single('image'), (req, res) => { const { artist, title } = req.body; const filename = req.file.filename; const artworks = JSON.parse(fs.readFileSync(artworksPath, 'utf-8')); artworks.push({ artist, title, filename }); fs.writeFileSync(artworksPath, JSON.stringify(artworks, null, 2)); res.redirect('/gallery.html'); }); app.get('/artworks', (req, res) => { const artworks = JSON.parse(fs.readFileSync(artworksPath, 'utf-8')); res.json(artworks); }); app.get('/download/:filename', (req, res) => { const file = path.join(__dirname, 'public', 'uploads', req.params.filename); res.download(file); }); app.listen(PORT, () => console.log(Server running on http://localhost:${PORT})); [] []
